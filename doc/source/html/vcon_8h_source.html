271 {
272  l4_msg_regs_t *mr = l4_utcb_mr_u(utcb);
273  mr->mr[0] = L4_VCON_WRITE_OP;
274  mr->mr[1] = size;

278  2 + (size + sizeof(l4_umword_t) - 1) / sizeof(l4_umword_t),
279  0, L4_MSGTAG_SCHEDULE),
280  L4_IPC_NEVER);

310 L4_INLINE int
311 l4_vcon_read_with_flags_u(l4_cap_idx_t vcon, char *buf, int size,
312  l4_utcb_t *utcb) L4_NOTHROW
313 {
314  int ret, r;
315  l4_msg_regs_t *mr;
316 
317  if (size < 0)
318  return -L4_EINVAL;
319 
320  mr = l4_utcb_mr_u(utcb);
321  mr->mr[0] = (size << 16) | L4_VCON_READ_OP;
322 
323  ret = l4_error_u(l4_ipc_call(vcon, utcb,
324  l4_msgtag(L4_PROTO_LOG, 1, 0, 0),
325  L4_IPC_NEVER),
326  utcb);
327  if (ret < 0)
328  return ret;
329 
330  r = mr->mr[0] & L4_VCON_READ_SIZE_MASK;
331 
332  if (!(mr->mr[0] & L4_VCON_READ_STAT_DONE)) // !eof
333  ret = size + 1;
334  else if (r < size)
335  ret = r;
336  else
337  ret = size;
338 
339  __builtin_memcpy(buf, &mr->mr[1], r < size ? r : size);
340  return ret | (mr->mr[0] & ~(L4_VCON_READ_STAT_DONE | L4_VCON_READ_SIZE_MASK));
341 }
