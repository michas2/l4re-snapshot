260 L4_INLINE l4_msgtag_t
261 l4_factory_create_gate_u(l4_cap_idx_t factory,
262  l4_cap_idx_t target_cap,
263  l4_cap_idx_t thread_cap, l4_umword_t label,
264  l4_utcb_t *u) L4_NOTHROW
265 {
266  l4_msgtag_t t;
267  l4_msg_regs_t *v;
268  int items = 0;
269  t = l4_factory_create_start_u(0, target_cap, u);
270  l4_factory_create_add_uint_u(label, &t, u);
271  v = l4_utcb_mr_u(u);
272  if (!(thread_cap & L4_INVALID_CAP_BIT))
273  {
274  items = 1;
275  v->mr[3] = l4_map_obj_control(0,0);
276  v->mr[4] = l4_obj_fpage(thread_cap, 0, L4_FPAGE_RWX).raw;
277  }
278  t = l4_msgtag(l4_msgtag_label(t), l4_msgtag_words(t), items, l4_msgtag_flags(t));
279  return l4_factory_create_commit_u(factory, t, u);
280 }

357 L4_INLINE int
358 l4_factory_create_add_fpage_u(l4_fpage_t d, l4_msgtag_t *tag,
359  l4_utcb_t *u) L4_NOTHROW
360 {
361  l4_msg_regs_t *v = l4_utcb_mr_u(u);
362  int w = l4_msgtag_words(*tag);
363  if (w + 2 > L4_UTCB_GENERIC_DATA_SIZE)
364  return 0;
365  v->mr[w] = L4_VARG_TYPE_FPAGE | (sizeof(l4_fpage_t) << 16);
366  v->mr[w + 1] = d.raw;
367  w += 2;
368  tag->raw = (tag->raw & ~0x3fUL) | (w & 0x3f);
369  return 1;
370 }

409 L4_INLINE int
410 l4_factory_create_add_lstr_u(char const *s, int len, l4_msgtag_t *tag,
411  l4_utcb_t *u) L4_NOTHROW
412 {
413 
414  l4_msg_regs_t *v = l4_utcb_mr_u(u);
415  int w = l4_msgtag_words(*tag);
416  char *c;
417  int i;
418 
419  if (w + 1 + (len + sizeof(l4_umword_t) - 1) / sizeof(l4_umword_t)
420  > L4_UTCB_GENERIC_DATA_SIZE)
421  return 0;
422 
423  v->mr[w] = L4_VARG_TYPE_STRING | (len << 16);
424  c = (char*)&v->mr[w + 1];
425  for (i = 0; i < len; ++i)
426  *c++ = *s++;
427 
428  w = w + 1 + (len + sizeof(l4_umword_t) - 1) / sizeof(l4_umword_t);
429 
430  tag->raw = (tag->raw & ~0x3fUL) | (w & 0x3f);
431  return 1;
432 }
