67 /* atomic compare and exchange 64 bit value */
68 #define __L4UTIL_ATOMIC_HAVE_ARCH_CMPXCHG64
69 L4_INLINE int
70 l4util_cmpxchg64(volatile l4_uint64_t * dest,
71  l4_uint64_t cmp_val, l4_uint64_t new_val)
72 {
73  l4_uint64_t tmp;
74 
75  __asm__ __volatile__
76  (
77  LOCK "cmpxchgq %1, %3 \n\t"
78  :
79  "=a" (tmp) /* 0 RAX, return val */
80  :
81  "r" (new_val), /* 1 reg, new value */
82  "0" (cmp_val), /* 2 RAX, compare value */
83  "m" (*dest) /* 3 mem, destination operand */
84  :
85  "memory", "cc"
86  );
87 
88  return tmp == cmp_val;
89 }
