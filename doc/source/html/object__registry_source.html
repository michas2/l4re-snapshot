99  int _register_gate(L4::Epiface *o, L4::Cap<L4::Ipc_gate> gate,
100  Demand const &demand)
101  {
102  int err = _sif->alloc_buffer_demand(demand);
103  if (err < 0)
104  return err;
105 
106  l4_umword_t id = l4_umword_t(o);
107  err = l4_error(gate->bind_thread(_server, id));

109  return err;

128  L4::Cap<void> _register_gate(L4::Epiface *o, Demand const &demand)
129  {
130  int err = _sif->alloc_buffer_demand(demand);
131  if (err < 0)
132  return L4::Cap<void>(err | L4_INVALID_CAP_BIT);
133 
134  Auto_cap<L4::Kobject>::Cap cap
135  = cap_alloc.alloc<L4::Kobject>();

137  if (!cap.is_valid())

279  bool unregister_obj(L4::Epiface *o)
280  {
281  if (!o || !o->obj_cap().is_valid())
282  return false;
283 
284  // make sure unhandled ipc ends up with the null handler
285  L4::Thread::Modify_senders todo;
286  todo.add(~3UL, reinterpret_cast<l4_umword_t>(o),
287  ~0UL, reinterpret_cast<l4_umword_t>((L4::Epiface*)&_null_handler));
288  _server->modify_senders(todo);
289 
290  L4::Cap<L4::Task>(L4Re::This_task)->unmap(o->obj_cap().fpage(),
291  L4_FP_ALL_SPACES);
292  o->set_server(0, L4::Cap<void>::Invalid);
293  return true;
